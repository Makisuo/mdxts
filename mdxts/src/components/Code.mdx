## Code Blocks

````mdx
```tsx filename="useCounter.ts"
import { useState } from 'react'

export function useCounter(initialValue: number = 0) {
  const [count, setCount] = useState(initialValue)
  return {
    count,
    increment: () => setCount(count + 1),
    decrement: () => setCount(count - 1),
  }
}
```

```tsx filename="Counter.tsx"
'use client'
import { useCounter } from './useCounter'

export default function Counter({
  initialCount = 0,
}: {
  initialCount: number
}) {
  const { count, decrement, increment } = useCounter(initialCount)
  return (
    <div>
      <button onClick={decrement}>-</button>
      <span>{count}</span>
      <button onClick={increment}>+</button>
    </div>
  )
}
```

```tsx
<Counter initialCount={2} />
```
````

Pass a `source` prop to the `Code` component to render a code block from a file. For example, the following MDX:

```mdx filename="Counter.mdx"
<Code source="./counter/Counter.tsx" />
```

Will produce a code block with the contents of `./counter/Counter.tsx`:

<Code allowErrors source="./counter/Counter.tsx" />
